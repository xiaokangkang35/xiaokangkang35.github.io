# 规则解析器

## JA000000011000000
```sql
SELECT
	a.rule_id,
	a.rule_name,
	A.RULE_DSCR,
	a.RULE_SOUC,
	VOLA_EVID,
	a.prog_code,
	b.ITEM_CODE,
	b.ITEM_NAME,
	b.sex,
	(SELECT replace_with_json(KNG_CONT, memo) AS replaced_text FROM rcc_rule_kng_cfg_a t WHERE t.rule_id = a.rule_id AND kng_CODE = 'TEXT_OUTPUT') 
FROM
	rcc_rule_d a,
	(
	SELECT
		RULE_ID,
		GROUP_CONCAT( CASE WHEN kng_code = 'ITEM_CODE' THEN KNG_CONT END ) ITEM_CODE,
		GROUP_CONCAT( CASE WHEN kng_code = 'ITEM_NAME' THEN KNG_CONT END ) ITEM_NAME ,
		GROUP_CONCAT( CASE WHEN kng_code = 'sex' THEN KNG_CONT END ) sex
	FROM
		rcc_rule_selfdef_kng_cfg_a  
	WHERE
		RULE_ID  in (select d.rule_id from rcc_dev_v2.rcc_rule_d d  where d.PROG_CODE='JA000000011000000' and d.rcc_rule_stas ='1'  )
	GROUP BY
		RULE_ID,
		kng_grp_id 
	) b
WHERE
	a.rule_id = b.rule_id
	AND A.RULE_ID in (select d.rule_id from rcc_rule_d d  where d.PROG_CODE='JA000000011000000' and  d.rcc_rule_stas='1')
```
